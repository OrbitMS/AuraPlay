import { listen, emit } from '@tauri-apps/api/event'; let player; // Inject YouTube API const tag = document.createElement('script'); tag.src = "https://www.youtube.com/iframe_api"; document.head.appendChild(tag); window.onYouTubeIframeAPIReady = () => { player = new YT.Player('yt-player', { height: '1', width: '1', events: { 'onReady': () => { listen('audio-control', (e) => { const { command, videoId, volume } = e.payload; if (command === 'play') player.loadVideoById(videoId); else if (command === 'pause') player.pauseVideo(); else if (command === 'resume') player.playVideo(); else if (command === 'set_volume') player.setVolume(volume); }); }, 'onStateChange': (event) => { if (event.data === YT.PlayerState.PLAYING) emit('audio-status', { state: 'playing' }); if (event.data === YT.PlayerState.PAUSED) emit('audio-status', { state: 'paused' }); } } }); };
